import numpy as np
import pandas as pd

import matplotlib.pyplot as plt

dataTab=['https://raw.githubusercontent.com/googletrends/data/master/20161205HollandeFrance.csv']


#7nl,just a fonction to help me format the Time
def formatDateTime(x):
    day=x.str.slice(0,2)
    month=x.str.slice(3,5)
    hour=x.str.slice(6,14)
    #print(month+"-"+day+" "+hour)
    x="2016-"+month+"-"+day+" "+hour
    return x

#
data= pd.read_csv(dataTab[0],header=1,encoding='latin1')

#transform de Time column to fit the timestamp format
data["Time"]=data["Time"].str.replace("-"," ")
data["Time"]=data["Time"].str.replace("h",":00:000000+0100")
data["Time"]=data["Time"].str.replace("/","-")
data["Time"]=formatDateTime(data["Time"])
data['Time']=pd.to_datetime(data['Time'])
data = data.sort_values('Time', ascending=True)

#Change the 'ç' to simple 'c' in "Francois Hollande"
#data.columns.values[1] = 'Francois Hollande'

#
fig = plt.figure(figsize=(12, 12))
people=['François Hollande','Emmanuel Macron','Arnaud Montebourg']

for nb in range(2,4):
    ax= fig.add_subplot(2,2,nb+1)
    ax.plot(data['Time'],data[data.columns.values[nb]])
    ax.set_title("pour "+data.columns.values[nb])
    plt.xticks(rotation=-45)

plt.show()
